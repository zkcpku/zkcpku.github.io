# time travel).

def is_gap(year):
    if year % 400 == 0:
        return True
    else:
        if year % 100 != 0 and year % 4 == 0:
            return True
    return False
 
def dayTo1(year, month, day):
    daysOfMonths = [ 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
    if is_gap(year):
        daysOfMonths[1] = 29
    now_month = 0
    result = 0
    while now_month < month-1:
        result = result + daysOfMonths[now_month]
        now_month = now_month + 1
    result =result + day
    return result
